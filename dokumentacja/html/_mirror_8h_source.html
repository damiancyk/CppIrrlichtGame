00127         f32 acos_f ( f32 deg )
00128         {
00129             // Fast computation of acos
00130             deg = ( f32 ) fmodf ( deg, 360 );
00131 
00132             if ( deg < 0 )
00133                 deg += 360;
00134 
00135             s32 deg_int = ( s32 ) deg;
00136 
00137             f32 deg_frac = deg - deg_int;
00138 
00139             return ( facos[deg_int] + deg_frac* ( facos[deg_int+1] - facos[deg_int] ) );

00437         // function used to create the view matrix. Provided by FreakNigh.
00438         matrix4 create_glfrustumLH ( f32 left, f32 right, f32 bottom, f32 top, f32 nearZ, f32 farZ )
00439         {
00440             f32 x = ( 2.0F * nearZ ) / ( right - left );
00441             f32 y = ( 2.0F * nearZ ) / ( top - bottom );
00442             f32 a = ( right + left ) / ( right - left );
00443             f32 b = ( top + bottom ) / ( top - bottom );
00444             f32 c = - ( farZ + nearZ ) / ( farZ - nearZ );
00445             f32 d = - ( 2.0F * farZ * nearZ ) / ( farZ - nearZ );
00446 
00447             matrix4 m;
00448             m.makeIdentity();
00449 #define M(row,col)  m[col*4+row]
00450             M ( 0, 0 ) = x;     M ( 0, 1 ) = 0.0F;  M ( 0, 2 ) = a;      M ( 0, 3 ) = 0.0F;
00451             M ( 1, 0 ) = 0.0F;  M ( 1, 1 ) = y;     M ( 1, 2 ) = b;      M ( 1, 3 ) = 0.0F;
00452             M ( 2, 0 ) = 0.0F;  M ( 2, 1 ) = 0.0F;  M ( 2, 2 ) = -c;      M ( 2, 3 ) = d;
00453             M ( 3, 0 ) = 0.0F;  M ( 3, 1 ) = 0.0F;  M ( 3, 2 ) = 1.0F;  M ( 3, 3 ) = 0.0F;
00454 #undef M
00455 
00456             return m;
00457         }
